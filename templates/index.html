// app.py — Flask receives and routes
@app.route("/chat", methods=["POST"])
def chat():
    data    = request.get_json()
    message = data.get("message", "").lower()
    
    # Pass to ML model
    intent, confidence = model.predict(message)
    response           = model.get_response(intent, message)
    
    # Send back to browser as JSON
    return jsonify({
        "intent":     intent,       # "report_defect"
        "confidence": confidence,   # 72.4
        "response":   response      # "Defect saved! Report ID: #4821..."
    })
# ML model classifies the intent
def predict(self, command):
    intent     = self.pipeline.predict([command])[0]
    proba      = self.pipeline.predict_proba([command])[0]
    confidence = max(proba)
    # Normalize: (raw - 1/n_classes) / (1 - 1/n_classes)
    n = len(self.pipeline.classes_)
    normalized = (confidence - 1/n) / (1 - 1/n)
    return intent, round(normalized * 100, 1)
    # Returns: ("report_defect", 72.4)
